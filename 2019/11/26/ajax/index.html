var char = newDate.getDate();
var xml = null;
if(window.XMLHttpRequest){
xml = new XMLHttpRequest();
}else{
new ActiveXObject(‘Microsoft.XMLHTTP’)
}

xml.open('GET', 'getNews.php?username=cg&age=100&date=' +char, true);

xml.send();

xml.onreadystatechange = function () {
    if(xml.readyState === 4){
        if(xml.status === 200){
            JSON.parse(xml.responseText)
        }
    }
}
