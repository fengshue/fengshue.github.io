var tool = require(‘./tool.js’);

module.exports = obj1;
var obj = {
init: function () {
//入口函数
var oDiv1 = tool.getDom(‘div1’);
this.bindEvent(oDiv1)
},

bindEvent: function (dom) {
    //绑定函数
    dom.onclick = this.changeColor;
},
changeColor: function () {
    this.style.backgroundColor = 'blue';
}

}

module.exports = obj;
